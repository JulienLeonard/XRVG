263   # build a color vector from hsv parametrization (convert from hsv to rgb) h, s, v being between 0.0 and 1.0
264   #
265   # taken from wikipedia[http://en.wikipedia.org/wiki/HSV_color_space]
266   #
267   # error on algo with h = 1.0 => hi == 6 must be taken into account
268   def Color.hsva( h, s, v, a)
269     h = Range.O.trim( h )
270     s = Range.O.trim( s )
271     v = Range.O.trim( v )
272     a = Range.O.trim( a )
273     values = Color.hsv(h,s,v) + [a]
274     return Color[*values]
275   end

277   def Color.hsv( h, s, v)
278     if s == 0.0
279       return [v, v, v]
280     end
281     h *= 360.0
282     hi = (h/60.0).floor
283     if hi == 6
284       hi = 5
285     end
286     f  = (h/60.0) - hi
287     p  = v * ( 1 -             s )
288     q  = v * ( 1 -       f   * s )
289     t  = v * ( 1 - ( 1 - f ) * s )
290     if hi == 0 
291       return [ v, t, p] 
292     end
293     if hi == 1 
294       return [ q, v, p] 
295     end
296     if hi == 2 
297       return [ p, v, t] 
298     end
299     if hi == 3 
300       return [ p, q, v] 
301     end
302     if hi == 4 
303       return [ t, p, v] 
304     end
305     if hi == 5 
306       return [ v, p, q] 
307     end
308   end

311   def Color.getHSLcomponent( tC, p, q ) #:nodoc:
312     while tC < 0.0
313       tC = tC + 1.0
314     end
315     while tC > 1.0
316       tC = tC - 1.0
317     end
318 
319     if tC < (1.0 / 6.0)
320       tC = p + ( (q-p) * 6.0 * tC )
321     elsif tC >=(1.0 / 6.0) and tC < 0.5
322       tC = q
323     elsif tC >= 0.5 and tC < (2.0 / 3.0)
324       tC = p + ( (q-p) * 6.0 * ((2.0 / 3.0) - tC) )
325     else
326       tC = p
327     end
328     return tC
329   end

343   def Color.hsl( h, s, l)
344     h *= 360.0
345     if l < 0.5
346       q = l * (1.0 + s)
347     else
348       q = l+ s - (l * s)
349     end
350     p = 2 * l - q
351     hk = h / 360.0
352     tR = hk + 1.0 / 3.0
353     tG = hk
354     tB = hk - 1.0 / 3.0
355 
356     tR = self.getHSLcomponent( tR, p, q )
357     tG = self.getHSLcomponent( tG, p, q )
358     tB = self.getHSLcomponent( tB, p, q )
359     return [tR, tG, tB]
360   end

362   # from http://en.wikipedia.org/wiki/HSL_and_HSV#Conversion_from_RGB_to_HSL_or_HSV
363   def Color.rgb2h(r,g,b)
364     result = 0.0
365     range = [r,g,b].range
366     if range.begin == range.end
367       result = 0.0
368     elsif range.end == r
369       result = (60.0 * (g - b) / range.size +   0.0)
370     elsif range.end == g
371       result = (60.0 * (b - r) / range.size + 120.0)
372     else
373       result = (60.0 * (r - g) / range.size + 240.0)
374     end
375     return (result % 360.0) / 360.0
376   end

378   def Color.rgb2sl(r,g,b)
379     range = [r,g,b].range
380     l = range.middle
381     s = 0.0
382     if range.begin == range.end
383       s = 0.0
384     elsif l <= 0.5
385       s = range.size / (2.0 * l)
386     else
387       s = range.size / (2.0 - 2.0 * l)
388     end
389     return [s,l]
390   end

392   def Color.rgb2sv(r,g,b)
393     range = [r,g,b].range
394     v = range.end
395     if v == 0.0
396       s = 0.0
397     else
398       s = (1.0 - (range.begin/range.end))
399     end
400     return [s,v]
401   end

427 # = Intro
428 # Palette defines color palettes, as interpolation between color points. As such, use Interpolation module, so uses for the moment only linear interpolation.
429 # But once built with interpolation, palette provides a continuous color "interval", and so is Samplable !
430 # = Use
431 #  palette  = Palette[ :colorlist, [ 0.0, Color.blue, 0.5, Color.orange, 1.0, Color.yellow ] ]
432 #  palette.rand( 10 )   # => return 10 random colors in palette
433 #  palette.color( 0.5 ) # => Color.orange

448   # build an interpolator by color componant
449   def build_interpolators()
450     vlists = [[],[],[],[]]
451     self.colorlist.foreach do |index, color|
452       values = []
453       if self.interpoldomain == :rgb
454 	values = color.rgba
455       elsif self.interpoldomain == :hsv
456 	values = color.hsva
457       elsif self.interpoldomain == :hsl
458 	values = color.hsla
459       else
460 	raise RuntimeError.new("#{self.interpoldomain} is an unknown color scheme.")
461       end
462       vlists[0] += [index, values[0] ]
463       vlists[1] += [index, values[1] ]
464       vlists[2] += [index, values[2] ]
465       vlists[3] += [index, values[3] ]
466     end
467     @interpolators = vlists.map {|samplelist| Interpolator[ :samplelist, samplelist, :interpoltype, self.interpoltype]}
468   end
469 
470   # interpolators accessor (for debugging)
471   def interpolators()
472     return @interpolators
473   end
474 
475   # overloading to reset interpolators if interpoltype changes
476   def interpoltype=(value)
477     @interpoltype = value
478     if @interpolators
479       self.build_interpolators
480     end
481   end

483   # method overloading to delegate computation to componant interpolators
484   def interpolate( dindex )
485     vs = self.interpolators.map {|inter| inter.interpolate( dindex )}
486     if self.interpoldomain == :rgb
487       return Color[ *vs ]
488     elsif self.interpoldomain == :hsv
489       return Color.hsva( *vs )
490     else
491       return Color.hsla( *vs )
492     end
493   end
