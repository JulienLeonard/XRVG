107 # = Example
108 #  quad = BinaryTree.new( [0.0,1.0, 0.2,0.0, 0.6,1.0,  0.8,0.0, 1.0,1.0]  )

112   def initialize( samplelist ) #:nodoc:
113     quads = []
114     ends  = []
115     samplelist.foreach(2).pairs do |ppair, pair|
116       pindex, pvalue = ppair
117       index, value   = pair
118       quads << BinaryTreeRange.new( nil, nil, nil, [pindex, pvalue, index, value] )
119       ends  << index
120     end

122   end

125     newquads = []
