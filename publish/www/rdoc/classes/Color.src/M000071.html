  def Color.hsv( h, s, v, a)
    if s == 0.0
      return Color[v, v, v, a]
    end
    h *= 360.0
    hi = (h/60.0).floor
    f  = (h/60.0) - hi
    p  = v * ( 1 -             s )
    q  = v * ( 1 -       f   * s )
    t  = v * ( 1 - ( 1 - f ) * s )
    if hi == 0 
      return Color[ v, t, p, a] 
    end
    if hi == 1 
      return Color[ q, v, p, a] 
    end
    if hi == 2 
      return Color[ p, v, t, a] 
    end
    if hi == 3 
      return Color[ p, q, v, a] 
    end
    if hi == 4 
      return Color[ t, p, v, a] 
    end
    if hi == 5 
      return Color[ v, p, q, a] 
    end
  end
