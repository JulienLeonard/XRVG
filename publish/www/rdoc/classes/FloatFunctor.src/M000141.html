  def apply( data, type )
    @applyhash = self.applyhash
    if not @applyhash.key? type
      Kernel::Raise("FloatFunctor::apply no regsitration for type #{type} and  object #{object.inspect}")
    else
      return self.send(@applyhash[type], data )
    end
  end
