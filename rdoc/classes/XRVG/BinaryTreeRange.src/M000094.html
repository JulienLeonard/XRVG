  def range( index )
    # Trace("BinaryTreeRange range index #{index} @limit #{@limit} quadright #{@quadright} quadleft #{@quadleft}")
    if @limit
      if index < @limit
        return @quadleft.range( index )
      else
        return @quadright.range( index )
      end
    else
      return @range
    end
  end
