  def Color.hsl( h, s, l)
    h *= 360.0
    if l < 0.5
      q = l * (1.0 + s)
    else
      q = l+ s - (l * s)
    end
    p = 2 * l - q
    hk = h / 360.0
    tR = hk + 1.0 / 3.0
    tG = hk
    tB = hk - 1.0 / 3.0

    tR = self.getHSLcomponent( tR, p, q )
    tG = self.getHSLcomponent( tG, p, q )
    tB = self.getHSLcomponent( tB, p, q )
    return [tR, tG, tB]
  end
