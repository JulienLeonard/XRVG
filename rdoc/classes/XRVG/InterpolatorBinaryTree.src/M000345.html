  def interpolate( dindex )
    if not @quad
      compute_quad
    end
    pindex, pvalue, index, value = @quad.range( dindex )
    if pindex.fequal?(index)
      result = value
    else
      result = pvalue + ((value + pvalue * (-1.0) ) * ((dindex - pindex) / (index - pindex )))
    end
    return result
  end
