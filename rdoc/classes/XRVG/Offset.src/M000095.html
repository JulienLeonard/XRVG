  def points
    result = []
    SyncS[self.abscissasampler, self.ampl].samples( self.nsamples) do |abscissa, amplsample|
      frame = self.support.frame( abscissa )
      result << frame.center + frame.vector.ortho.norm * amplsample
    end
    return result
  end
