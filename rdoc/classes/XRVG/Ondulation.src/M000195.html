  def compute
    abscissas = self.abscissasampler.samples( self.freq + 1 )
    sens = 1.0
    pieces = []
    [abscissas.pairs, self.ampl.samples( self.freq )].forzip do |abspair, amplitude|
      abs1, abs2 = abspair
      pieces += self.compute_interpol( abs1, abs2, amplitude, sens )
      sens *= -1.0
    end
    return pieces
  end
