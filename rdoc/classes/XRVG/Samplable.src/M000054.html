  def apply_samples( inputs, &block )
    if not block
      return inputs.map {|abs| self.apply_sample( abs ) }
    else
      if inputs.length > 0
        container = self.apply_sample( inputs[0] ); yield container;
        inputs[1..-1].each {|abs| yield self.apply_sample( abs, container ) }
      end
    end
  end
