  def compute
    result = []
    self.curvesampler.samples( self.nmotifs + 1).pairs do |p1,p2|
      # Trace("SimilarMotifIterator::compute p1 #{p1.inspect} p2 #{p2.inspect}")
      newbezier = self.motif.similar( (p1..p2) )
      result += newbezier.data
    end
    return result
  end
