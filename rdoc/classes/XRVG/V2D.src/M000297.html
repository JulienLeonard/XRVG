  def coords(type=:cartesien)
    if type == :cartesien
      return [self.x, self.y]
    elsif type == :polar
      if @polarcoords == nil
        @polarcoords = [self.r, self.angle]
      end
      return @polarcoords
    else
      Kernel.raise( "Unknown coord type #{type}" )      
    end
  end
