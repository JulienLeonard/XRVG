  9 #
 10 # Base module to define float lists processing and computation.
 11 # = Design principle
 12 # Composite design pattern, to be able to chain functors
 13 # = Concepts
 14 # on each functor object :
 15 # - "trigger" method triggers float processing, that is 
 16 #   * if arg is integer, call "generate" to get values
 17 #   * else (array float values), just keep them
 18 # - pass inputs to "process" method, which wraps "compute" method to be able to call block and iterate
 19 # - "compute" method first call "modify" which is the recursive processing function
 20 # - then call "apply" which is specific processing method of the functor
 21 # - "apply" then dispatchs processing according to the processing "type", ie "sample" or "split"
 22 # - by default, FloatFunctor provides "transforms" and "transform" to be used by "type" processing
 23 module FloatFunctor

355 # Filter class
356 # = Intro
357 # Filter class allows to call generically a method with arg between 0.0..0.1 on a given object, for use in a +FloatFunctor+ context.
358 # = Example
359 # this allow to do for example 
360 #  SyncS[bezier1.filter(:point), bezier1.filter(:tangent), palette].samples( 10 ) do |point, tangent, color|
361 class Filter
362   include Samplable
363   
364   # to define a filter on "object" with method "samplemethod", or with a block (exclusive)
365   #
366   # from these two alternatives, returns a proc to call for float processing
367   def initialize(object=nil, samplemethod=:sample, &block)
368     if object
369       # Trace("Filter::initialize object #{object.inspect} method #{samplemethod.inspect} ")
370       @proc = object.method( samplemethod )
371     else
372       @proc = Proc.new( &block )
373     end
374   end
375 
376   # Samplable redefinition : call previous proc built from initialize to process float "abs"
377   def transform( abs )
378     result = @proc.call( abs )
379     # Trace("Filter::transform abs #{abs} result #{result.inspect}")
380     return result
381   end

393 # = RandomFilter class, to resample randomly
394 # == Attributes
395 #   attribute :mindiff, 0.0
396 #   attribute :sort, false
397 # :mindiff specifies which minimum interval must be preserved between random values. Of course, the number of sampling must then be above a given value so that this constraint can be verified.
398 class RandomFilter < Filter
399   include Attributable
400   attribute :mindiff, 0.0
401   attribute :sort, false
402   attribute :withboundaries, false
403 
404   # make sampling by trying to check :mindiff constraint
405   # 
406   # generate an exception if not possible
407   def rawtransforms(nsamples)
408     size  = 1.0
409     nsplit = nsamples - 1
410     rsize = size - nsplit * @mindiff
411     if rsize < 0.0
412       raise("RandomFilter rawtransforms error: nsamples #{nsamples} mindiff #{@mindiff} are incompatible")
413     end
414     
415     mindiffs = Array.new( nsplit, @mindiff )
416     
417     # compute now nsplit values whose sum is <= to rsize
418     randarray = [0.0]
419     subarray = (0.0..rsize).rand( nsplit-1 )
420     if not subarray.is_a? Array
421       subarray = [subarray]
422     end
423     randarray += subarray
424     randarray.push( rsize )
425     randarray.sort!
426 
427     rsizes = Array.new
428     randarray.each_cons(2) { |min, max| rsizes.push( (0.0..max-min).rand ) }
429 
430     rsum = rsizes.sum
431     root = (0.0..(rsize-rsum)).rand
432 
433     Trace("mindiffs #{mindiffs.inspect} rsizes #{rsizes.inspect} rsize #{rsize} rsum #{rsum} root #{root}")
434 
435     preresult = []
436     mindiffs.zip( rsizes ) {|mindiff, rsize| preresult.push( mindiff + rsize )}
437 
438     result = [root]
439     preresult.each {|v| newv = result[-1] + v; result << newv }
440 
441     return result
442   end
443   
444   # do (0.0..1.0).rand to speed up computation if @mindiff is 0.0
445   def basictransform( v )
446     return (0.0..1.0).rand
447   end
448 
449   def transforms( inputs, type )  #:nodoc:
450     if type == :split
451       @sort = true
452     end
453     if @mindiff == 0.0
454       result = inputs.map { |v| self.basictransform( v ) }
455       if @sort
456 	result = result.sort
457       end
458       if type == :split
459 	result[0]  = 0.0
460 	result[-1] = 1.0
461       end
462     else
463       result = self.rawtransforms( inputs.size )
464       if not @sort
465 	result = result.shuffle
466       end
467     end
468     if self.withboundaries
469       result[0] = inputs[0]
470       result[-1] = inputs[-1]
471     end
472     return result
473   end
